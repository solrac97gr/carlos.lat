	type Config struct {

	type Service struct {

	}

	func (s *Service) GetAPIKey() string {

	}

	func (s *Service) LoadConfigJSON() error {

		configFile, err := os.Open("./config.json")

		}

		if err = jsonParser.Decode(&configuration); err != nil {

		}
		s.Config = configuration
		return nil
	}

	func NewConfigService() *Service {
		return &Service{}
	}
